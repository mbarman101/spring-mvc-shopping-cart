 USE cart_db;

CREATE TABLE USER_INFO (
  id INT(11) NOT NULL AUTO_INCREMENT,
  NAME VARCHAR(45) NOT NULL,
  email VARCHAR(45) NOT NULL,
  user_role VARCHAR(45) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE ITEM_INFO (
  id INT(11) NOT NULL AUTO_INCREMENT,
  NAME VARCHAR(45) NOT NULL,
  price INT(45) NOT NULL,
  PRIMARY KEY (id)
);
